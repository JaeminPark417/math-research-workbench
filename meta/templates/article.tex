\begin{theorem}
For every real number $x$, one has $x^2\geq 0$.
\end{theorem}

\begin{proof}
If $x\geq 0$, then $x^2\geq 0$. If $x<0$, then $-x>0$ and
$x^2=(-x)^2\geq 0$.
\end{proof}
